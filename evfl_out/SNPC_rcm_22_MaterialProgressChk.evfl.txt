flow Chk_HowManyType:

        entry_variables['MaterialRemain'] += 1
    if event_flags_system['cLand:RcmShopMaterial_SoftWood']:
        entry_variables['MaterialRemain'] += 1
    if event_flags_system['cLand:RcmShopMaterial_Iron']:
        entry_variables['MaterialRemain'] += 1
    if entry_variables['MaterialRemain'] != 0:
        subflow_results[0] = 0
    elif entry_variables['MaterialRemain'] != 1:
        subflow_results[0] = 1

        subflow_results[0] = 2
