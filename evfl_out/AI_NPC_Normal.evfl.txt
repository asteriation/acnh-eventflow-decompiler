flow State_Wait:
    MainNpc.NpcChangeAIState('cWait', '', '', '', false, true)
    return

flow State_Talk:
    MainNpc.NpcChangeAIState('cTalk', '', '', '', false, true)
    return

flow State_TalkCast:
    MainNpc.NpcChangeAIState('cTalkCast', '', '', '', false, true)
    return

flow State_Speak:
    MainNpc.NpcChangeAIState('cSpeak', '', '', '', false, true)
    return

flow State_Wander:
    MainNpc.NpcChangeAIState('cWander', '', '', '', false, true)
    return

flow State_Conversation:
    MainNpc.NpcChangeAIState('cConversation', '', '', '', false, true)
    return

flow End_Hit:
    if (MainNpc.NpcIsHitByNet()) and (MainNpc.NpcIsFleaHolder()):
        run State_Speak
        run sub_Event15
    else:
        if MainNpc.NpcHitNum(3):
            run State_Speak
            run sub_Event15
        else:
            MainNpc.NpcSetHitTimer(60)
            run Root
    return

flow State_Hit:
    MainNpc.NpcChangeAIState('cHit', '', '', '', false, true)
    return

flow Push:
    run State_Speak
    return

flow StartTalk:
    run State_Talk
    return

flow BecomeTalkCast:
    run State_TalkCast
    return

flow EndTalk:
    return

flow Root:
    run State_Wait
    return

flow sub_Event15:
    MainNpc.NpcClearHitParam()
