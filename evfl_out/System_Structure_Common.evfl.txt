flow PreviewStructure:
    EventFlowSystemActor.FadeOut('cNormalFader', 'cNormalFader', 'cWhite', 0.5, 0.5, true)
    Player.PlayerChangeDemoState('cSharePlaySubHideUntilDemoEnd')
    EventFlowSystemActor.StructureDeletePreviewGuide(false)
    EventFlowSystemActor.PreviewStructure('StructureType', 'StructurePosType', 'BuildTiming')
    EventFlowSystemActor.FadeIn(true)
    return
