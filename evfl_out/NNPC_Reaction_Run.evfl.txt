flow Root:
    if (EventFlowSystemActor.IsInsectChasingPlayer(0)) or (EventFlowSystemActor.IsInsectChasingPlayer(1)) or (EventFlowSystemActor.IsInsectChasingPlayer(2)):
        MainNpc.NpcSetReactionType('cRun')
        if event_flags_npc['cNpcTemp:ContinuousReactionTalkCount'] > 1:
            fEvent3 = EventFlowSystemActor.RandomChoiceExcludePrevious3(true, 3)
            if fEvent3 == 0:
                event_flags_npc['cNpcTemp:ReactionTalkLabel'] = 0
            elif fEvent3 == 1:
                event_flags_npc['cNpcTemp:ReactionTalkLabel'] = 1
            elif fEvent3 == 2:
                event_flags_npc['cNpcTemp:ReactionTalkLabel'] = 2
        MainNpc.SetItemName(0, 15, 2)
        if event_flags_npc['cNpcTemp:ReactionTalkLabel'] == 0:
            if event_flags_npc['cNpcTemp:ReactionTalkLabel'] == 1:
                run Run_03
            else:
                run Run_02
        else:
            run Run_01
        MainNpc.SetNpcFeel('cNormal', 120)
        event_flags_npc['cNpcMemory:AcquaintanceshipFlagOff'] = true
        event_flags_npc['cNpcTemp:ReserveWarningBee'] = true
        EventFlowSystemActor.ExitFlowchart()
    return
