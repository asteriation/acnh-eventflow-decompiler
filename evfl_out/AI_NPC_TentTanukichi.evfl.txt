flow Root:
    MainNpc.NpcChangeSpNpcTurnState('cSpNpcTurn', 'Root', 'SNPC_rco', '', true, true, 30, 2, 0, 0, true, false, '', '')
    return

flow Init:
    MainNpc.NpcRegistAIBlackBoard('CheckRobResult', 0)
    if EventFlowSystemActor.SystemCheckNowStage('cIdrTanukichi'):
        MainNpc.NpcAISetCheckRobFunc('Ftr_Working', 'Root_Tutorial', 'CheckRob', 'cTanukichiTent', false, false)
    MainNpc.NpcAISetting(0, true)
    MainNpc.NpcAISetting(1, true)
    MainNpc.NpcAISetting(8, true)
    return

flow CheckRob:
    if not event_flags_system['cPlayer:MakeVillagePlayerFlag']:
        MainNpc.NpcSetAIBlackBoard('CheckRobResult', 0)
    elif event_flags_system['cPlayer:RcoFinishMakingAxe']:
        MainNpc.NpcSetAIBlackBoard('CheckRobResult', 0)
    else:
        MainNpc.NpcSetAIBlackBoard('CheckRobResult', 1)
    return
