flow Init:
    MainNpc.NpcAISetting(1, true)
    if event_flags_system['cLandTemp:PrologueStartedCampfire']:
        MainNpc.NpcRegistAIBlackBoard('IsForceLife', 1)
        MainNpc.NpcRegistAIBlackBoard('PrologueStep', 2)
    elif event_flags_system['cLand:SetPrologueBonfire']:
        MainNpc.NpcRegistAIBlackBoard('IsForceLife', 0)
        MainNpc.NpcRegistAIBlackBoard('PrologueStep', 2)
    else:
        MainNpc.NpcRegistAIBlackBoard('PrologueStep', 0)
        MainNpc.NpcRegistAIBlackBoard('IsForceLife', 0)
    MainNpc.NpcRegistAIBlackBoard('BuildingSiteIndex', -1)
    MainNpc.NpcRegistAIBlackBoard('HasTent', 1)
    MainNpc.NpcRegistAIBlackBoard('IsMove', 0)
    MainNpc.NpcRegistAIBlackBoard('ArrivePlaza', 0)
    return

flow Root:
    run AI_NPC_Prologue_Common::Root_Common
    return
