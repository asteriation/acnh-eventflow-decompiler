flow FreeV_Chk:
    if EventFlowSystemActor.SystemCheckNowStage('cTalkingNpcHouse'):
        run NNPC_Free_HouseV::Root
    elif EventFlowSystemActor.SystemCheckNowStage('cAnyNpcHouse'):
        run NNPC_Free_House::Free_HouseG
    else:
        run NNPC_Free_FieldV::Root
    return

flow Free_Chk:
    if EventFlowSystemActor.SystemCheckNowStage('cTalkingNpcHouse'):
        if MainNpc.NpcIsExistOtherNpc():
            run NNPC_Free_House::Free_HouseH
        else:
            run NNPC_Free_House::Root
    elif EventFlowSystemActor.SystemCheckNowStage('cAnyNpcHouse'):
        run NNPC_Free_House::Free_HouseG
    else:
        if EventFlowSystemActor.SystemCheckNowStage('cTalkingPlayerHouse'):
            run NNPC_Free_House::Root
        else:
            run NNPC_Free_Field::Root
    return

flow Root:
    if EventFlowSystemActor.NetIsConnected():
        run FreeV_Chk
    else:
        run Free_Chk
    return
