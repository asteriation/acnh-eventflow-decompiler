flow MainUI:
    EventFlowSystemActor.UINetConnectWaitDisappear()
    EventFlowSystemActor.UIMyDesignExchangeMenuHandling()
    fEvent12 = EventFlowSystemActor.MyDesignExchangeMenuResult()
    if fEvent12 == 0:
        run Seq_Close
    elif fEvent12 == 3:
        run Obj_WorldMyDesignGallery_03_Upload::Root
    elif fEvent12 == 4:
        run Obj_WorldMyDesignGallery_05_MyOpus::Root
    elif fEvent12 == 2:
        run Obj_WorldMyDesignGallery_04_Search::SearchByAuthorID
    elif fEvent12 == 5:
        run Obj_WorldMyDesignGallery_06_Search2::SearchByCondition
    elif fEvent12 == 6:
        run Obj_WorldMyDesignGallery_06_Search2::SearchByFavoriteAuthor
    elif fEvent12 == 1:
        EventFlowSystemActor.MyDesignExchangeResetError()
        run Obj_WorldMyDesignGallery_04_Search::SearchByMyDesignID
    return

flow Root:
    EventFlowSystemActor.SystemFaderRelease()
    EventFlowSystemActor.WaitFrame(30)
    if EventFlowSystemActor.NetIsDreaming():
        run Obj_WorldMyDesignGallery_10_InDream::SearchByAuthorID

    return

    EventFlowSystemActor.MessageSuspend()
    EventFlowSystemActor.WaitFrame(15)
    if EventFlowSystemActor.NetIsDreaming():
        EventFlowSystemActor.CameraReserveFirstStateCommand('Talk:MyDesignExchangeInDream')
    else:
        EventFlowSystemActor.CameraReserveFirstStateCommand('Talk:MyDesignExchange')
    EventFlowSystemActor.ReturnStage('', 'cMyDesignBack', 'cMyDesignBack', 'cBlack', 0.5, 0.5)
    return
