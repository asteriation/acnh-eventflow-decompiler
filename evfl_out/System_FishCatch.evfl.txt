flow StartTimer:
    EventFlowSystemActor.BGMPropertyControl(5)
    EventFlowSystemActor.UIMiniGameTimerAppear(180, 'cFishFes')
    return

flow BgmStop:
    EventFlowSystemActor.BGMPropertyControl(2)
    EventFlowSystemActor.DisappearChatBaloon('cBaloonType_TimerEvent')
    if EventFlowSystemActor.IsMyVillage():
        event_flags_system['cPlayer:BevQuestFinish'] = true
    else:
        event_flags_system['cPlayerVisit:DC_BevQuestFinish'] = true
    return

flow FinishTimer:
    EventFlowSystemActor.BGMPropertyControl(3)
    if EventFlowSystemActor.IsSharePlay(0):
        EventFlowSystemActor.SetSharePlayChangeCondition(false)
    return
